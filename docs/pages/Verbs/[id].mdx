import graph from '../../data/schema.org.types.json'
import Link from 'next/link'

export const getStaticProps = async ({params}) => {

  const { id } = params

  const schema = graph.find(item => `${item.label}Action` === id)

  return {
    props: {
      schema
    }
  }
}

export const getStaticPaths = async () => {

  const ids = graph.map(schema => schema.label)
  const paths = ids.filter(item => item.match(/.+Action/))
                   .map(item => ({ params: { id: item === 'Action' ? item : item.replace('Action', '') } }))

  console.log(paths)

  return {
    paths,
    fallback: false
  }
}
